Preparar la validación cruzada


# 1. Configuración de la Validación Cruzada de 10 pliegues (10-fold CV)
# Nota: Usamos el método estándar "cv" para evaluar de forma robusta
cv_driver <- trainControl(method = "cv", number = 10, verboseIter = FALSE)

Entrenamiento de las tres regresiones lineales (simple, múltiple y)

# =========================================================================
# ENFOQUE A: Regresión Lineal Simple 
# Usamos 'median_income' porque justificamos que tiene la mayor correlación (0.69)
# =========================================================================
modelo_simple <- train(
  median_house_value ~ median_income, 
  data = data_train,
  method = "lm",
  metric = "RMSE",
  trControl = cv_driver
)

# =========================================================================
# ENFOQUE B: Regresión Lineal Múltiple
# Usamos el punto (.) para indicarle a R que incluya TODOS los predictores
# =========================================================================
modelo_multiple <- train(
  median_house_value ~ ., 
  data = data_train,
  method = "lm",
  metric = "RMSE",
  trControl = cv_driver
)

# =========================================================================
# ENFOQUE C: Stepwise Regression (Selección secuencial óptima por AIC)
# =========================================================================
# Definimos los modelos límites para el algoritmo stepwise
modelo_minimo <- lm(median_house_value ~ 1, data = data_train)
modelo_maximo <- lm(median_house_value ~ ., data = data_train)

# El algoritmo evalúa qué variables quitar o poner de forma automática
step_formula <- step(
  modelo_minimo, 
  scope = list(lower = modelo_minimo, upper = modelo_maximo), 
  direction = "both", 
  trace = FALSE
)

# Guardamos el resultado del Stepwise dentro de caret para calcular su CV
modelo_stepwise <- train(
  step_formula$call$formula,
  data = data_train,
  method = "lm",
  metric = "RMSE",
  trControl = cv_driver
)

Asignación de modelo champion

# Modelo ganador basado en la tabla de resultados
champion_model <- modelo_multiple

#  Generar predicciones sobre el set de prueba y calcular el RMSE final
preds_champion <- predict(champion_model, newdata = data_test)
rmse_test <- sqrt(mean((preds_champion - data_test$median_house_value)^2))

# Calcular el porcentaje de error respecto al rango de entrenamiento
rango_train <- max(data_train$median_house_value) - min(data_train$median_house_value)
porcentaje_error <- (rmse_test / rango_train) * 100
